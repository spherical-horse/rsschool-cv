    function sortArray(array) {
      const res = [];
      let oddCounter = 0;
      const odds = array.filter(x => x % 2 !== 0).sort((a, b) => a - b);
      for (let i = 0; i < array.length; i++){
        if (array[i] % 2 === 0){
          res.push(array[i]);
        } else {
          res.push(odds[oddCounter]);
          oddCounter++;
        }
      }
      return res;
    }
